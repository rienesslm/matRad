179%% Treatment Plan
180% The next step is to define your treatment plan labeled as 'pln'. This 
181% structure requires input from the treatment planner and defines the most
182% important cornerstones of your treatment plan.
183%%
184% First of all, we need to define what kind of radiation modality we would
185% like to use. Possible values are photons, protons or carbon. In this
186% example we would like to use photons for treatment planning. Next, we
187% need to define a treatment machine to correctly load the corresponding 
188% base data. matRad features generic base data in the file
189% 'photons_Generic.mat'; consequently the machine has to be set to 'Generic'
190pln.radiationMode = 'photons';            
191pln.machine       = 'Generic';
192
193%%
194% Define the flavor of biological optimization for treatment planning along
195% with the quantity that should be used for optimization. Possible values 
196% are (none: physical dose based optimization; const_RBExD: constant RBE of 1.1; 
197% LEMIV_effect: effect-based optimization; LEMIV_RBExD: optimization of 
198% RBE-weighted dose. As we use photons, we select 'none' as we want to optimize the 
199% physical dose.
200pln.propOpt.bioOptimization = 'none';                                              
201
202%%
203% The remaining plan parameters are set like in the previous example files
204pln.numOfFractions        = 30;
205pln.propStf.gantryAngles  = [0 45];
206pln.propStf.couchAngles   = [0 0];
207pln.propStf.bixelWidth    = 5;
208pln.propStf.numOfBeams    = numel(pln.propStf.gantryAngles);
209pln.propStf.isoCenter     = ones(pln.propStf.numOfBeams,1) * matRad_getIsoCenter(cst,ct,0);
210pln.propOpt.runDAO        = 0;
211pln.propOpt.runSequencing = 0;

213% dose calculation settings

218%% Generate Beam Geometry STF

221%% Dose Calculation
222dij = matRad_calcPhotonDose(ct,stf,pln,cst);

224%% Inverse Optimization for intensity-modulated photon therapy
225% The goal of the fluence optimization is to find a set of bixel/spot 
226% weights which yield the best possible dose distribution according to the
227% clinical objectives and constraints underlying the radiation treatment.
228resultGUI = matRad_fluenceOptimization(dij,cst,pln);
229
230%% Plot the resulting dose slice
231plane      = 3;

233doseWindow = [0 max([resultGUI.physicalDose(:)])];
234
235figure,title('phantom plan')
236matRad_plotSliceWrapper(gca,ct,cst,1,resultGUI.physicalDose,plane,slice,[],[],colorcube,[],doseWindow,[]);
