31%% Treatment Plan
32% The next step is to define your treatment plan labeled as 'pln'. This 
33% structure requires input from the treatment planner and defines the most
34% important cornerstones of your treatment plan.
35
36pln.radiationMode  = 'photons';  
37pln.machine        = 'Generic';
38pln.numOfFractions = 30;
39pln.propOpt.bioOptimization = 'none';    
40pln.propStf.gantryAngles    = [0];
41pln.propStf.couchAngles     = [0];
42pln.propStf.bixelWidth      = 10;
43pln.propStf.numOfBeams      = numel(pln.propStf.gantryAngles);
44pln.propStf.isoCenter       = ones(pln.propStf.numOfBeams,1) * matRad_getIsoCenter(cst,ct,0);
45pln.propOpt.runSequencing   = 0;
46pln.propOpt.runDAO          = 0;
47
48% dose calculation settings
49pln.propDoseCalc.doseGrid.resolution.x = 3; % [mm]
50pln.propDoseCalc.doseGrid.resolution.y = 3; % [mm]
51pln.propDoseCalc.doseGrid.resolution.z = 3; % [mm]
52
53%% Generate Beam Geometry STF
54stf = matRad_generateStf(ct,cst,pln);
55
56%% Dose Calculation
57% Calculate dose influence matrix for unit pencil beam intensities using

59dij = matRad_calcPhotonDoseMC(ct,stf,pln,cst);

61%% Inverse Optimization for IMRT
62resultGUI = matRad_fluenceOptimization(dij,cst,pln);
63
64%% Plot the Resulting Dose Slice

67figure,
68imagesc(resultGUI.physicalDose(:,:,slice)),colorbar, colormap(jet)

70%%
71% Exemplary, we show how to obtain the dose in the target and plot the histogram
