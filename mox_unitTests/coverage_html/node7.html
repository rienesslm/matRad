52%% Prepare Metadata
53%if the field is not set, we assume standard matRad x-y swap
54if ~isfield(metadata,'axisPermutation')

59%use the matrad coordinate system
60if ~isfield(metadata,'coordinateSystem')

63%If there is no image origin set, center the image
64imageExtent = metadata.resolution .* size(cube);

68%we can also store the center
69if ~isfield(metadata,'imageCenter')
