23%% Patient Data Import
24% Let's begin with a clear Matlab environment. Then, import the TG119
25% phantom into your workspace. The phantom is comprised of a 'ct' and 'cst'
26% structure defining the CT images and the structure set. Make sure the 
27% matRad root directory with all its subdirectories is added to the Matlab 
28% search path.
29
30matRad_rc; %If this throws an error, run it from the parent directory first to set the paths
31
32load('TG119.mat');
33
34%%
35% The file TG119.mat contains two Matlab variables. Let's check what we 
36% have just imported. First, the 'ct' variable comprises the ct cube along
37%with some meta information describing properties of the ct cube (cube 
38% dimensions, resolution, number of CT scenarios). Please note that 
39%multiple ct cubes (e.g. 4D CT) can be stored in the cell array ct.cube{}
40display(ct);
41
42%%
43% The 'cst' cell array defines volumes of interests along with information 
44% required for optimization. Each row belongs to one certain volume of 
45% interest (VOI), whereas each column defines different properties. 
46% Specifically, the second and third column  show the name and the type of 
47% the structure. The type can be set to OAR, TARGET or IGNORED. The fourth 
48% column contains a linear index vector that lists all voxels belonging to 
49% a certain VOI.
50display(cst);
51
52%%
53% The fifth column represents meta parameters for optimization. The overlap
54% priority is used to resolve ambiguities of overlapping structures (voxels 
55% belonging to multiple structures will only be assigned to the VOI(s) with
56% the highest overlap priority, i.e.. the lowest value). The parameters 
57% alphaX and betaX correspond to the tissue's photon-radiosensitivity 
58% parameter of the linear quadratic model. These parameter are required for
59% biological treatment planning using a variable RBE. Let's output the meta 
60% optimization parameter of the target, which is stored in the thrid row:
61ixTarget = 3;
62display(cst{ixTarget,5});
63
64%%
65% The sixth column contains optimization information such as objectives and
66% constraints which are required to calculate the objective function value. 
67% Please note, that multiple objectives/constraints can be defined for
68% individual structures. Here, we have defined a squared deviation 
69% objective making it 'expensive/costly' for the optimizer to over- and 
70% underdose the target structure (both are equally important). 
71display(cst{ixTarget,6});
72
73%% Treatment Plan
74% The next step is to define your treatment plan labeled as 'pln'. This 
75% matlab structure requires input from the treatment planner and defines 
76% the most important cornerstones of your treatment plan.
77
78%%
79% First of all, we need to define what kind of radiation modality we would
80% like to use. Possible values are photons, protons or carbon. In this case
81% we want to use photons. Then, we need to define a treatment machine to 
82% correctly load the corresponding base data. matRad includes base data for
83% generic photon linear accelerator called 'Generic'. By this means matRad 
84% will look for 'photons_Generic.mat' in our root directory and will use 
85% the data provided in there for dose calculation
86pln.radiationMode = 'photons';  
87pln.machine       = 'Generic';
88
89%%
90% Define the flavor of optimization along with the quantity that should be
91% used for optimization. Possible values are (none: physical optimization; 
92% const_RBExD: constant RBE of 1.1; LEMIV_effect: effect-based 
93% optimization; LEMIV_RBExD: optimization of RBE-weighted dose. As we are 
94% using photons, we simply set the parameter to 'none' thereby indicating 
95% the physical dose should be optimized.
96pln.propOpt.bioOptimization = 'none';    
97
98%%
99% Now we have to set some beam parameters. We can define multiple beam 
100% angles for the treatment and pass these to the plan as a vector. matRad 
101% will then interpret the vector as multiple beams. In this case, we define
102% linear spaced beams from 0 degree to 359 degree in 40 degree steps. This 
103% results in 9 beams. All corresponding couch angles are set to 0 at this 
104% point. Moreover, we set the bixelWidth to 5, which results in a beamlet 
105% size of 5 x 5 mm in the isocenter plane. The number of fractions is set 
106% to 30. Internally, matRad considers the fraction dose for optimization, 
107% however, objetives and constraints are defined for the entire treatment.
108pln.numOfFractions         = 30;
109pln.propStf.gantryAngles   = [0:40:359];
110pln.propStf.couchAngles    = zeros(1,numel(pln.propStf.gantryAngles));
111pln.propStf.bixelWidth     = 5;
112
113%%
114% Obtain the number of beams and voxels from the existing variables and 
115% calculate the iso-center which is per default the center of gravity of 
116% all target voxels.
117pln.propStf.numOfBeams      = numel(pln.propStf.gantryAngles);
118pln.propStf.isoCenter       = ones(pln.propStf.numOfBeams,1) * matRad_getIsoCenter(cst,ct,0);
119
120%% dose calculation settings
121% set resolution of dose calculation and optimization
122pln.propDoseCalc.doseGrid.resolution.x = 3; % [mm]
123pln.propDoseCalc.doseGrid.resolution.y = 3; % [mm]
124pln.propDoseCalc.doseGrid.resolution.z = 3; % [mm]
125
126%%
127% Enable sequencing and disable direct aperture optimization (DAO) for now.
128% A DAO optimization is shown in a seperate example.
129pln.propOpt.runSequencing = 1;
130pln.propOpt.runDAO        = 0;
131
132%%
133% and et voila our treatment plan structure is ready. Lets have a look:
134display(pln);
135
136%% Generate Beam Geometry STF
137% The steering file struct comprises the complete beam geometry along with 
138% ray position, pencil beam positions and energies, source to axis distance (SAD) etc.
139stf = matRad_generateStf(ct,cst,pln);
140
141%%
142% Let's display the beam geometry information of the 6th beam
143display(stf(6));
144
145%% Dose Calculation
146% Let's generate dosimetric information by pre-computing dose influence 
147% matrices for unit beamlet intensities. Having dose influences available 
148% allows subsequent inverse optimization.
149dij = matRad_calcPhotonDose(ct,stf,pln,cst);
150
151%% Inverse Optimization for IMRT
152% The goal of the fluence optimization is to find a set of beamlet/pencil 
153% beam weights which yield the best possible dose distribution according to
154% the clinical objectives and constraints underlying the radiation 
155% treatment. Once the optimization has finished, trigger once the GUI to 
156% visualize the optimized dose cubes.
157resultGUI = matRad_fluenceOptimization(dij,cst,pln);
158matRadGUI;
159
160%% Plot the Resulting Dose Slice
161% Let's plot the transversal iso-center dose slice
162slice = round(pln.propStf.isoCenter(1,3)./ct.resolution.z);
163figure
164imagesc(resultGUI.physicalDose(:,:,slice)),colorbar, colormap(jet);
165
166%% Now let's create another treatment plan but this time use a coarser beam spacing.
167% Instead of 40 degree spacing use a 50 degree geantry beam spacing
168pln.propStf.gantryAngles = [0:50:359];
169pln.propStf.couchAngles  = zeros(1,numel(pln.propStf.gantryAngles));
170pln.propStf.numOfBeams   = numel(pln.propStf.gantryAngles);
171stf                      = matRad_generateStf(ct,cst,pln);
172pln.propStf.isoCenter    = vertcat(stf.isoCenter);
173dij                      = matRad_calcPhotonDose(ct,stf,pln,cst);
174resultGUI_coarse         = matRad_fluenceOptimization(dij,cst,pln);
175
176%%  Visual Comparison of results
177% Let's compare the new recalculation against the optimization result.
178% Check if you have added all subdirectories to the Matlab search path,
179% otherwise it will not find the plotting function
180plane      = 3;
181doseWindow = [0 max([resultGUI.physicalDose(:); resultGUI_coarse.physicalDose(:)])];
182
183figure,title('original plan - fine beam spacing')
184matRad_plotSliceWrapper(gca,ct,cst,1,resultGUI.physicalDose,plane,slice,[],0.75,colorcube,[],doseWindow,[]);
185figure,title('modified plan - coarse beam spacing')
186matRad_plotSliceWrapper(gca,ct,cst,1,resultGUI_coarse.physicalDose,plane,slice,[],0.75,colorcube,[],doseWindow,[]);
187
188%% 
189% At this point we would like to see the absolute difference of the first 
190% optimization (finer beam spacing) and the second optimization (coarser 
191% beam spacing)
192absDiffCube = resultGUI.physicalDose-resultGUI_coarse.physicalDose;
193figure,title( 'fine beam spacing plan - coarse beam spacing plan')
194matRad_plotSliceWrapper(gca,ct,cst,1,absDiffCube,plane,slice,[],[],colorcube);
195
196%% Obtain dose statistics
197% Two more columns will be added to the cst structure depicting the DVH and
198% standard dose statistics such as D95,D98, mean dose, max dose etc.
199[dvh,qi]               = matRad_indicatorWrapper(cst,pln,resultGUI);
200[dvh_coarse,qi_coarse] = matRad_indicatorWrapper(cst,pln,resultGUI_coarse);
201
202%%
203% The treatment plan using more beams should in principle result in a
204% better OAR sparing. Therefore lets have a look at the D95 of the OAR of 
205% both plans
206ixOAR = 2;
207display(qi(ixOAR).D_95);
208display(qi_coarse(ixOAR).D_95);
209
210
211%% 
212% Export the dose in binary format. matRad can write multiple formats, here we 
213% choose to export in nrrd format using the matRad_writeCube function, which 
214% chooses the appropriate subroutine from the extension. The metadata struct can
215% also store more optional parameters, but requires only the resolution to be s
216% set.
217metadata = struct('resolution',[ct.resolution.x ct.resolution.y ct.resolution.z]);
218matRad_writeCube([pwd filesep 'photonDose_example2.nrrd'],resultGUI.physicalDose,'double',metadata);
