49% Add isocenter to source and target point. Because the algorithm does not
50% works with negatives values. This put (resolution.x,resolution.y,resolution.z)
51% in the center of first voxel
52
53sourcePoint = sourcePoint + isocenter;
54targetPoint = targetPoint + isocenter;
55
56% Save the numbers of planes.
57[yNumPlanes, xNumPlanes, zNumPlanes] = size(cubes{1});
58xNumPlanes = xNumPlanes + 1;
59yNumPlanes = yNumPlanes + 1;
60zNumPlanes = zNumPlanes + 1;
61
62% eq 11
63% Calculate the distance from source to target point.
64d12 = norm(sourcePoint-targetPoint);
65
66% eq 3
67% Position of first planes in millimeter. 0.5 because the central position
68% of the first voxel is at [resolution.x resolution.y resolution.z]
69xPlane_1 = .5*resolution.x;
70yPlane_1 = .5*resolution.y;
71zPlane_1 = .5*resolution.z;
72
73% Position of last planes in milimiter
74xPlane_end = (xNumPlanes - .5)*resolution.x;
75yPlane_end = (yNumPlanes - .5)*resolution.y;
76zPlane_end = (zNumPlanes - .5)*resolution.z;
77
78% eq 4
79% Calculate parametrics values of \alpha_{min} and \alpha_{max} for every
80% axis, intersecting the ray with the sides of the CT. 
81if targetPoint(1) ~= sourcePoint(1)
82    aX_1   = (xPlane_1 - sourcePoint(1)) / (targetPoint(1) - sourcePoint(1));
83    aX_end = (xPlane_end - sourcePoint(1)) / (targetPoint(1) - sourcePoint(1));
84else
85    aX_1   = [];
86    aX_end = [];
87end
88if targetPoint(2) ~= sourcePoint(2)
89    aY_1   = (yPlane_1 - sourcePoint(2)) / (targetPoint(2) - sourcePoint(2));
90    aY_end = (yPlane_end - sourcePoint(2)) / (targetPoint(2) - sourcePoint(2));
91else
92    aY_1   = [];
93    aY_end = [];
94end
95if targetPoint(3) ~= sourcePoint(3)
96    aZ_1   = (zPlane_1 - sourcePoint(3)) / (targetPoint(3) - sourcePoint(3));
97    aZ_end = (zPlane_end - sourcePoint(3)) / (targetPoint(3) - sourcePoint(3));
98else
99    aZ_1   = [];
100    aZ_end = [];
101end
102
103% eq 5
104% Compute the \alpha_{min} and \alpha_{max} in terms of parametric values
105% given by equation 4.
106alpha_min = max([0 min(aX_1,aX_end) min(aY_1,aY_end) min(aZ_1,aZ_end)]);
107alpha_max = min([1 max(aX_1,aX_end) max(aY_1,aY_end) max(aZ_1,aZ_end)]);
108
109% eq 6
110% Calculate the range of indeces who gives parametric values for
111% intersected planes.
112if targetPoint(1) == sourcePoint(1)
113    i_min = []; i_max = [];
114elseif targetPoint(1) > sourcePoint(1)
115    i_min = xNumPlanes - (xPlane_end - alpha_min * (targetPoint(1) - sourcePoint(1)) - sourcePoint(1))/resolution.x;
116    i_max = 1          + (sourcePoint(1) + alpha_max * (targetPoint(1) - sourcePoint(1)) - xPlane_1)/resolution.x;
117    % rounding
118    i_min = ceil(1/1000*(round(1000*i_min)));
119    i_max = floor(1/1000*(round(1000*i_max)));
120else
121    i_min = xNumPlanes - (xPlane_end - alpha_max * (targetPoint(1) - sourcePoint(1)) - sourcePoint(1))/resolution.x;
122    i_max = 1          + (sourcePoint(1) + alpha_min * (targetPoint(1) - sourcePoint(1)) - xPlane_1)/resolution.x;
123    i_min = ceil(1/1000*(round(1000*i_min)));
124    i_max = floor(1/1000*(round(1000*i_max)));
125end
126if targetPoint(2) == sourcePoint(2)
127    j_min = []; j_max = [];
128elseif targetPoint(2) > sourcePoint(2)
129    j_min = yNumPlanes - (yPlane_end - alpha_min * (targetPoint(2) - sourcePoint(2)) - sourcePoint(2))/resolution.y;
130    j_max = 1          + (sourcePoint(2) + alpha_max * (targetPoint(2) - sourcePoint(2)) - yPlane_1)/resolution.y;
131    j_min = ceil(1/1000*(round(1000*j_min)));
132    j_max = floor(1/1000*(round(1000*j_max)));
133else
134    j_min = yNumPlanes - (yPlane_end - alpha_max * (targetPoint(2) - sourcePoint(2)) - sourcePoint(2))/resolution.y;
135    j_max = 1          + (sourcePoint(2) + alpha_min * (targetPoint(2) - sourcePoint(2)) - yPlane_1)/resolution.y;
136    j_min = ceil(1/1000*(round(1000*j_min)));
137    j_max = floor(1/1000*(round(1000*j_max)));
138end
139if targetPoint(3) == sourcePoint(3)
140    k_min = []; k_max = [];
141elseif targetPoint(3) >= sourcePoint(3)
142    k_min = zNumPlanes - (zPlane_end - alpha_min * (targetPoint(3) - sourcePoint(3)) - sourcePoint(3))/resolution.z;
143    k_max = 1          + (sourcePoint(3) + alpha_max * (targetPoint(3) - sourcePoint(3)) - zPlane_1)/resolution.z;
144    k_min = ceil(1/1000*(round(1000*k_min)));
145    k_max = floor(1/1000*(round(1000*k_max)));
146else
147    k_min = zNumPlanes - (zPlane_end - alpha_max * (targetPoint(3) - sourcePoint(3)) - sourcePoint(3))/resolution.z;
148    k_max = 1          + (sourcePoint(3) + alpha_min * (targetPoint(3) - sourcePoint(3)) - zPlane_1)/resolution.z;
149    k_min = ceil(1/1000*(round(1000*k_min)));
150    k_max = floor(1/1000*(round(1000*k_max)));
151end
152
153% eq 7
154% For the given range of indices, calculate the paremetrics values who
155% represents intersections of the ray with the plane.
156if i_min ~= i_max
157    if targetPoint(1) > sourcePoint(1)
158        alpha_x = (resolution.x*(i_min:1:i_max)-sourcePoint(1)-.5*resolution.x)/(targetPoint(1)-sourcePoint(1));
159    else
160        alpha_x = (resolution.x*(i_max:-1:i_min)-sourcePoint(1)-.5*resolution.x)/(targetPoint(1)-sourcePoint(1));
161    end
162else
163    alpha_x = [];
164end
165if j_min ~= j_max
166    if targetPoint(2) > sourcePoint(2)
167        alpha_y = (resolution.y*(j_min:1:j_max)-sourcePoint(2)-.5*resolution.y)/(targetPoint(2)-sourcePoint(2));
168    else
169        alpha_y = (resolution.y*(j_max:-1:j_min)-sourcePoint(2)-.5*resolution.y)/(targetPoint(2)-sourcePoint(2));
170    end
171else
172    alpha_y = [];
173end
174if k_min ~= k_max
175    if targetPoint(3) > sourcePoint(3)
176        alpha_z = (resolution.z*(k_min:1:k_max)-sourcePoint(3)-.5*resolution.z)/(targetPoint(3)-sourcePoint(3));
177    else
178        alpha_z = (resolution.z*(k_max:-1:k_min)-sourcePoint(3)-.5*resolution.z)/(targetPoint(3)-sourcePoint(3));
179    end
180else
181    alpha_z = [];
182end
183
184% eq 8
185% Merge parametrics sets.
186alphas = unique([alpha_min alpha_x alpha_y alpha_z alpha_max]);
187
188% eq 10
189% Calculate the voxel intersection length.
190l = d12*diff(alphas);
191
192% eq 13
193% Calculate \alpha_{middle}
194alphas_mid = .5*(alphas(1:end-1)+alphas(2:end));
195
196% eq 12
197% Calculate the voxel indices: first convert to physical coords
198i_mm = sourcePoint(1) + alphas_mid*(targetPoint(1) - sourcePoint(1));
199j_mm = sourcePoint(2) + alphas_mid*(targetPoint(2) - sourcePoint(2));
200k_mm = sourcePoint(3) + alphas_mid*(targetPoint(3) - sourcePoint(3));
201% then convert to voxel index
202i = round(i_mm/resolution.x);
203j = round(j_mm/resolution.y);
204k = round(k_mm/resolution.z);
205
206% Handle numerical instabilities at the borders.
207i(i<=0) = 1; j(j<=0) = 1; k(k<=0) = 1;
208i(i>xNumPlanes-1) = xNumPlanes-1;
209j(j>yNumPlanes-1) = yNumPlanes-1;
210k(k>zNumPlanes-1) = zNumPlanes-1;
211
212% Convert to linear indices
213ix = j + (i-1)*size(cubes{1},1) + (k-1)*size(cubes{1},1)*size(cubes{1},2); 
214
215% obtains the values from cubes
216rho = cell(numel(cubes),1);
217for i = 1:numel(cubes)
218    rho{i} = cubes{i}(ix);
219end
