42% set up list with bev coordinates for calculation of radiological depth

46% calculate spacing of rays on ray matrix

49% define candidate ray matrix covering 1000x1000mm^2
50numOfCandidateRays = 2 * ceil(500/rayMxSpacing) + 1;
51candidateRayMx     = zeros(numOfCandidateRays);
52
53% define coordinates
54[candidateRaysCoords_X,candidateRaysCoords_Z] = meshgrid(rayMxSpacing*[floor(-500/rayMxSpacing):ceil(500/rayMxSpacing)]);
55
56% check which rays should be used
57for i = 1:stf.numOfRays
58   
59    ix = (candidateRaysCoords_X(:) - (1+rayMx_bev_y/stf.SAD)*stf.ray(i).rayPos_bev(1)).^2 + ...

61           <= lateralCutoff^2;
62    
63    candidateRayMx(ix) = 1;
64    
65end

67% set up ray matrix
68rayMx_bev = [candidateRaysCoords_X(logical(candidateRayMx(:))) ...
69             rayMx_bev_y*ones(sum(candidateRayMx(:)),1) ...  
70             candidateRaysCoords_Z(logical(candidateRayMx(:)))];
71
72%     figure,
73%     for jj = 1:length(rayMx_bev)

80% rotate ray matrix from bev to world coordinates

83% criterium for ray selection

87for i = 1:size(rayMx_world,1)

89    % run siddon ray tracing algorithm

105
106    ixRememberFromCurrTracing = x_dist > -raySelection & x_dist <= raySelection ...
107                              & z_dist > -raySelection & z_dist <= raySelection;
108
109    if any(ixRememberFromCurrTracing) > 0

111        for j = 1:ct.numOfCtScen

114            % eq 14

119            dCum = cumsum(d)-d/2;

122            radDepthCube{j}(ixHitVoxel(ixRememberFromCurrTracing)) = dCum(ixRememberFromCurrTracing);
123        end
124    end

126end
