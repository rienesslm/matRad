16%%
17% In this example we will show 
18% (i) how to load patient data into matRad
19% (ii) how to setup a proton dose calculation 
20% (iii) how to inversely optimize the pencil beam intensities directly from command window in MATLAB. 
21% (iv) how to simulate a lateral patient displacement by shifting the iso-center 
22% (v) how to recalculated the dose considering the shifted geometry and the previously optimized pencil beam intensities
23% (vi) how to compare the two results

25%% Patient Data Import
26% Let's begin with a clear Matlab environment and import the prostate
27% patient into your workspace
28
29matRad_rc; %If this throws an error, run it from the parent directory first to set the paths
30
31load('PROSTATE.mat');
32
33%% Treatment Plan
34% The next step is to define your treatment plan labeled as 'pln'. This 
35% structure requires input from the treatment planner and defines the most 
36% important cornerstones of your treatment plan.
37
38%%
39% First of all, we need to define what kind of radiation modality we would
40% like to use. Possible values are photons, protons or carbon. In this
41% example we would like to use protons for treatment planning. Next, we
42% need to define a treatment machine to correctly load the corresponding 
43% base data. matRad features generic base data in the file
44% 'proton_Generic.mat'; consequently the machine has to be set accordingly
45pln.radiationMode = 'protons';        
46pln.machine       = 'Generic';
47
48%%
49% Define the flavor of biological optimization for treatment planning along
50% with the quantity that should be used for optimization. Possible values 
51% are (none: physical optimization; const_RBExD: constant RBE of 1.1; 
52% LEMIV_effect: effect-based optimization; LEMIV_RBExD: optimization of 
53% RBE-weighted dose. As we use protons, we follow here the clinical 
54% standard and use a constant relative biological effectiveness of 1.1. 
55% Therefore we set bioOptimization to const_RBExD
56pln.propOpt.bioOptimization = 'const_RBExD';
57
58%%
59% for particles it is possible to also calculate the LET disutribution
60% alongside the physical dose. Therefore you need to activate the
61% corresponding option during dose calculcation
62pln.propDoseCalc.calcLET = 1;
63                                       
64%%
65% Now we have to set the remaining plan parameters.
66pln.numOfFractions        = 30;
67pln.propStf.gantryAngles  = [90 270];
68pln.propStf.couchAngles   = [0 0];
69pln.propStf.bixelWidth    = 3;
70pln.propStf.numOfBeams    = numel(pln.propStf.gantryAngles);
71pln.propStf.isoCenter     = ones(pln.propStf.numOfBeams,1) * matRad_getIsoCenter(cst,ct,0);
72pln.propOpt.runDAO        = 0;
73pln.propOpt.runSequencing = 0;
74
75% dose calculation settings
76pln.propDoseCalc.doseGrid.resolution.x = 3; % [mm]
77pln.propDoseCalc.doseGrid.resolution.y = 3; % [mm]
78pln.propDoseCalc.doseGrid.resolution.z = 3; % [mm]
79
80%% Generate Beam Geometry STF
81stf = matRad_generateStf(ct,cst,pln);
82
83%% Dose Calculation
84% Lets generate dosimetric information by pre-computing dose influence 
85% matrices for unit beamlet intensities. Having dose influences available 
86% allows for subsequent inverse optimization. 
87dij = matRad_calcParticleDose(ct,stf,pln,cst);
88
89%% Inverse Optimization for IMPT
90% The goal of the fluence optimization is to find a set of bixel/spot 
91% weights which yield the best possible dose distribution according to the 
92% clinical objectives and constraints underlying the radiation treatment
93resultGUI = matRad_fluenceOptimization(dij,cst,pln);
94
95%% Plot the Resulting Dose Slice
96% Let's plot the transversal iso-center dose slice
97slice = round(pln.propStf.isoCenter(1,3)./ct.resolution.z);
98figure
99imagesc(resultGUI.RBExDose(:,:,slice)),colorbar,colormap(jet)
100
101%% Plot the Resulting Beam Dose Slice
102% Let's plot the transversal iso-center dose slice of beam 1 and beam 2
103% separately 
104figure
105subplot(121),imagesc(resultGUI.RBExDose_beam1(:,:,slice)),colorbar,colormap(jet),title('dose of beam 1')
106subplot(122),imagesc(resultGUI.RBExDose_beam2(:,:,slice)),colorbar,colormap(jet),title('dose of beam 2')
107
108%% and the corresponding LET distribution
109% Transversal iso-center slice
110figure
111imagesc(resultGUI.LET(:,:,slice)),colormap(jet),colorbar,title('LET [keV/µm]')
112
113%%
114% Now let's simulate a patient shift in y direction for both beams
115stf(1).isoCenter(2) = stf(1).isoCenter(2) - 4;
116stf(2).isoCenter(2) = stf(2).isoCenter(2) - 4;
117pln.propStf.isoCenter       = reshape([stf.isoCenter],[3 pln.propStf.numOfBeams])';
118
119%% Recalculate Plan
120% Let's use the existing optimized pencil beam weights and recalculate the RBE weighted dose
121resultGUI_isoShift = matRad_calcDoseDirect(ct,stf,pln,cst,resultGUI.w);
122
123%%  Visual Comparison of results
124% Let's compare the new recalculation against the optimization result.
125plane = 3;
126doseWindow = [0 max([resultGUI.RBExDose(:); resultGUI_isoShift.RBExDose(:)])];
127
128figure,title('original plan')
129matRad_plotSliceWrapper(gca,ct,cst,1,resultGUI.RBExDose,plane,slice,[],0.75,colorcube,[],doseWindow,[]);
130figure,title('shifted plan')
131matRad_plotSliceWrapper(gca,ct,cst,1,resultGUI_isoShift.RBExDose,plane,slice,[],0.75,colorcube,[],doseWindow,[]);
132 
133absDiffCube = resultGUI.RBExDose-resultGUI_isoShift.RBExDose;
134figure,title('absolute difference')
135matRad_plotSliceWrapper(gca,ct,cst,1,absDiffCube,plane,slice,[],[],colorcube);
136
137% Let's plot single profiles that are perpendicular to the beam direction
138ixProfileY = round(pln.propStf.isoCenter(1,2)./ct.resolution.y);
139
140profileOrginal = resultGUI.RBExDose(:,ixProfileY,slice);
141profileShifted = resultGUI_isoShift.RBExDose(:,ixProfileY,slice);
142
143figure,plot(profileOrginal,'LineWidth',2),grid on,hold on, 
144       plot(profileShifted,'LineWidth',2),legend({'original profile','shifted profile'}),
145       xlabel('mm'),ylabel('Gy(RBE)'),title('profile plot')
146       
147%% Quantitative Comparison of results
148% Compare the two dose cubes using a gamma-index analysis. The gamma index
149% is a composite quality distribution equally taking into account a dose 
150% difference and a distance to agreement criterion in order to quantify
151% differences between two dose cubes. A gamma-index value of smaller than 1
152% indicates a successful test and a value greater than 1 illustrates a 
153% failed test.
154
155doseDifference  = 2;
156distToAgreement = 2;
157n               = 1;
158
159[gammaCube,gammaPassRateCell] = matRad_gammaIndex(...
160    resultGUI_isoShift.RBExDose,resultGUI.RBExDose,...
161    [ct.resolution.x, ct.resolution.y, ct.resolution.z],...
162    [doseDifference distToAgreement],slice,n,'global',cst);
163
164
165[env, ~] = matRad_getEnvironment();
166% Let's plot the gamma index histogram
167figure
168hist(gammaCube(gammaCube>0),100)
169title('gamma index histogram')
