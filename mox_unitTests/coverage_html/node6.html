304%This function wraps datatype definitions to the one used by MATLAB
305function datatype = getMATLABdataType(typestring)
306%The typedefinitions are taken directly from Section 5 of the NRRD
307%definition
308switch typestring
309    case {'signed char','int8','int8_t'}
310        datatype = 'int8';     
311    case {'uchar','unsigned char','uint8','uint8_t'}
312        datatype = 'uint8';        
313    case {'short','short int','signed short','signed short int','int16','int16_t'}
314        datatype = 'int16';        
315    case {'ushort','unsigned short','unsigned short int','uint16','uint16_t'}
316        datatype = 'uint16';        
317    case {'int','signed int','int32','int32_t'}
318        datatype = 'int32';        
319    case {'uint','unsigned int','uint32','uint32_t'}
320        datatype = 'uint32';        
321    case {'longlong','long long','long long int','signed long long','signed long long int','int64','int64_t'}
322        datatype = 'int64';        
323    case {'ulonglong', 'unsigned long long', 'unsigned long long int','uint64','uint64_t'}
324        datatype = 'uint64';        
325    case 'float'
326        datatype = 'single';        
327    case 'double'
328        datatype = 'double';        
329    otherwise
330        error('Could not identify datatype for NRRD data');
331end

333end
