65function dvh = getDVHPoints(cst, sIx, doseCube, dvhPoints, dvhType)
66n = numel(dvhPoints);
67dvh         = NaN * ones(1,n);
68indices     = cst{sIx,4}{1};
69numOfVoxels = numel(indices);
70
71doseInVoi   = doseCube(indices);
72
73switch dvhType
74    case 'cum' % cummulative DVH
75        for j = 1:n
76            dvh(j) = sum(doseInVoi >= dvhPoints(j));
77        end
78
79    case 'diff' % differential DVH
80        binning = (dvhPoints(2) - dvhPoints(1))/2;
81        for j = 1:n % differential DVH        
82            dvh(j) = sum(dvhPoints(j) + binning > doseInVoi & doseInVoi > dvhPoints(j) - binning);
83        end
84
85end
86dvh = dvh ./ numOfVoxels * 100;
