44% Define function_Di
45func_Di = @(beta,x) beta/(beta-m) * (exp(-m*x) - exp(-beta*x)); 
46
47% Calulate lateral distances using grid interpolation.
48lat1 = Interp_kernel1(isoLatDistsX,isoLatDistsZ);
49lat2 = Interp_kernel2(isoLatDistsX,isoLatDistsZ);
50lat3 = Interp_kernel3(isoLatDistsX,isoLatDistsZ);
51
52% now add everything together (eq 19 w/o inv sq corr -> see below)
53dose = lat1 .* func_Di(betas(1),radDepths) + ...
54       lat2 .* func_Di(betas(2),radDepths) + ...
55       lat3 .* func_Di(betas(3),radDepths);
56
57% inverse square correction
58dose = dose .* (SAD./geoDists(:)).^2;
59
60% check if we have valid dose values and adjust numerical instabilities
