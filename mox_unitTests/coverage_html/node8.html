1function matRad_writeMHA(filepath,cube,metadata)
2% matRad function to write mha files
3% 
4% call
5%   matRad_writeMHA(filepath,cube,metadata)
6%
7% input
8%   filepath:   full filename (with extension)
9%   cube:       3D array to be written into file
10%   metadata:   struct of metadata. Writer will wrap the existing metadata 
11%               to MHA standard-specific fields 
12%               Necessary fieldnames are:
13%               - resolution: [x y z]
14%               - datatype: numeric MATLAB-Datatype
15%
16% output
17%   file will be written to disk
18%

35%Sanity checks and restrictions
36dimensions = size(cube);
37if numel(dimensions) ~= 3
38    error('Sorry! matRad only supports 3-dimensional MHA output');
39end
40
41fid = fopen(filepath, 'wb');
42if fid <= 0
43    error('Could not open MHA destination file!');
44end
45cleaner = onCleanup(@() fclose(fid));
46
47%We perform the permutation
48if isfield(metadata,'axisPermutation')
49    cube = permute(cube,metadata.axisPermutation);
50end
51%The transformation matrix is now the unit matrix
52transformMatrix = diag(ones(1,numel(dimensions)));
53tmString = sprintf(' %d',transformMatrix(:));
54
55%Determine the endian
56[~,~,endian] = computer;
57switch endian
58    case 'L'
59        byteOrderMSB = 'True';
60    case 'B'
61        byteOrderMSB = 'False';
62    otherwise
63        error('Unknown endian!');
64end
65
66fprintf(fid, 'ObjectType = Image\n');
67fprintf(fid, 'NDims = %d\n',numel(dimensions));
68fprintf(fid, 'BinaryData = True\n');
69fprintf(fid, 'BinaryDataByteOrderMSB = %s\n',byteOrderMSB); %Not sure about this field
70fprintf(fid, 'ElementByteOrderMSB = %s\n',byteOrderMSB); %Not sure about this field
71fprintf(fid, 'TransformMatrix =%s\n',tmString);
72fprintf(fid, 'Offset = %f %f %f\n',metadata.imageOrigin(1),metadata.imageOrigin(2),metadata.imageOrigin(3));
73fprintf(fid, 'AnatomicalOrientation = RAI\n'); %Did not double check this line
74fprintf(fid, 'ElementSpacing = %f %f %f\n',metadata.resolution(1),metadata.resolution(2),metadata.resolution(3));
75fprintf(fid, 'DimSize = %d %d %d\n',dimensions(1),dimensions(2),dimensions(3));
76fprintf(fid, 'ElementType = %s\n',matlabTypeToMHAtype(metadata.datatype));
77fprintf(fid, 'ElementDataFile = LOCAL\n');
78fwrite(fid,cube,metadata.datatype,'b');
79fclose(fid);
80
81end

83function newType = matlabTypeToMHAtype(datatype)
84switch datatype
85    case {'single','float'}
86        newType = 'MET_FLOAT';     
87    case 'double'
88        newType = 'MET_DOUBLE';
89    case {'uchar','uint8'}
90       newType = 'MET_UCHAR';
91    case {'logical','int8','char'}
92        newType = 'MET_CHAR';
93    case 'int16'
94        newType = 'MET_SHORT';
95    case 'uint16'
96        newType = 'MET_USHORT';
97    case 'int32'
98        newType = 'MET_INT';
99    case 'int64'
100        newType = 'MET_LONG';
101    case 'uint32'
102        newType = 'MET_UINT';
103    case 'uint64'
104        newType = 'MET_ULONG';
105   otherwise
106       error(['Datatype ' datatype ' not supported by MHA exporter!']);
107end
108end
