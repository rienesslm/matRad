126function gain = mcSquare_magicFudge(energy)
127% mcSquare will scale the spot intensities in
128% https://gitlab.com/openmcsquare/MCsquare/blob/master/src/data_beam_model.c#L906
129% by this factor so we need to divide up front to make things work. The
130% original code can be found at https://gitlab.com/openmcsquare/MCsquare/blob/master/src/compute_beam_model.c#L16
131
132K = 35.87; % in eV (other value 34.23 ?)
133
134% // Air stopping power (fit ICRU) multiplied by air density
135SP = (9.6139e-9*energy^4 - 7.0508e-6*energy^3 + 2.0028e-3*energy^2 - 2.7615e-1*energy + 2.0082e1) * 1.20479E-3 * 1E6; % // in eV / cm
136
137% // Temp & Pressure correction
138PTP = 1.0; 
139
140% // MU calibration (1 MU = 3 nC/cm)
141% // 1cm de gap effectif
142C = 3.0E-9; % // in C / cm
143
144% // Gain: 1eV = 1.602176E-19 J
145gain = (C*K) / (SP*PTP*1.602176E-19);
146
147% divide by 1e7 to not get tiny numbers...
148gain = gain/1e7;
149
150end
