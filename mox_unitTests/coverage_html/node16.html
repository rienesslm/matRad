75            %displayToConsole lowest-level logging function for matRad.
76            %   Display to console will be called from the public wrapper
77            %   functions dispError, dispWarning, dispInfo, dispDebug
78            %
79            %  input

82            %    formatSpec:  string to print using format specifications similar to fprintf
83            %    varargin:    variables according to formatSpec
84            
85            if nargin < 4

87            else
88                forwardArgs = [{formatSpec},varargin(:)'];
89            end

91            if obj.keepLog
92                obj.messageLog{end+1,1} = upper(type);
93                obj.messageLog{end,2} = sprintf(forwardArgs{:});
94            end
95            
96            switch type
97                case{'info'}
98                    if obj.logLevel >= 3
99                        fprintf(forwardArgs{:});
100                    end
101                case{'debug'}
102                    if obj.logLevel >= 5
103                        forwardArgs{1} = ['DEBUG: ' forwardArgs{1}];
104                        fprintf(forwardArgs{:});
105                    end
106                case{'dep'}
107                    if obj.logLevel >= 4
108                        forwardArgs{1} = ['DEPRECATION WARNING: ' forwardArgs{1}];
109                        warning(forwardArgs{:});
110                    end
111                case{'warning'}
112                    if obj.logLevel >= 2
113                        warning(forwardArgs{:});
114                    end
115                case {'error'}

118                        %stack trace from the last two files/lines (i.e.,
119                        %this function / file)
120                        
121                        err.message = sprintf(forwardArgs{:});
122                        err.identifier = 'matRad:Error';
123                        err.stack = dbstack(2);
124                        error(err);
125                        
126                    end
127                otherwise
128                    error('Log type %s not defined!',type);
129            end

131            if obj.writeLog
