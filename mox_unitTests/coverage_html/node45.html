18% (i) how to load patient data into matRad

25%% Patient Data Import
26% Let's begin with a clear Matlab environment and import the head &
27% neck patient into your workspace.
28
29matRad_rc; %If this throws an error, run it from the parent directory first to set the paths
30
31load('HEAD_AND_NECK.mat');
32
33%% Treatment Plan
34% The next step is to define your treatment plan labeled as 'pln'. This 
35% structure requires input from the treatment planner and defines 
36% the most important cornerstones of your treatment plan.
37
38pln.radiationMode   = 'photons';   % either photons / protons / carbon
39pln.machine         = 'Generic';
40pln.numOfFractions  = 30;
41
42pln.propOpt.bioOptimization = 'none';    
43pln.propStf.gantryAngles    = [0:72:359];
44pln.propStf.couchAngles     = [0 0 0 0 0];
45pln.propStf.bixelWidth      = 5;
46pln.propStf.numOfBeams      = numel(pln.propStf.gantryAngles);
47pln.propStf.isoCenter       = ones(pln.propStf.numOfBeams,1) * matRad_getIsoCenter(cst,ct,0);
48
49% dose calculation settings
50pln.propDoseCalc.doseGrid.resolution.x = 3; % [mm]
51pln.propDoseCalc.doseGrid.resolution.y = 3; % [mm]
52pln.propDoseCalc.doseGrid.resolution.z = 3; % [mm]
53
54% We can also use other solver for optimization than IPOPT. matRad 
55% currently supports fmincon from the MATLAB Optimization Toolbox. First we
56% check if the fmincon-Solver is available, and if it es, we set in in the
57% pln.propOpt.optimizer vairable. Otherwise wie set to the default
58% optimizer 'IPOPT'
59if matRad_OptimizerFmincon.IsAvailable()
60    pln.propOpt.optimizer = 'fmincon';   
61else
62    pln.propOpt.optimizer = 'IPOPT';
63end
64
65%%
66% Enable sequencing and direct aperture optimization (DAO).
67pln.propOpt.runSequencing = 1;
68pln.propOpt.runDAO        = 1;
69
70%% Generate Beam Geometry STF
71stf = matRad_generateStf(ct,cst,pln);
72
73%% Dose Calculation
74% Lets generate dosimetric information by pre-computing dose influence 
75% matrices for unit beamlet intensities. Having dose influences available 
76% allows for subsequent inverse optimization.
77dij = matRad_calcPhotonDose(ct,stf,pln,cst);
78
79%% Inverse Planning for IMRT
80% The goal of the fluence optimization is to find a set of beamlet weights 
81% which yield the best possible dose distribution according to the 
82% predefined clinical objectives and constraints underlying the radiation 
83% treatment. Once the optimization has finished, trigger once the GUI to
84% visualize the optimized dose cubes.
85resultGUI = matRad_fluenceOptimization(dij,cst,pln);
86matRadGUI;
87
88%% Sequencing
89% This is a multileaf collimator leaf sequencing algorithm that is used in 
90% order to modulate the intensity of the beams with multiple static 
91% segments, so that translates each intensity map into a set of deliverable 
92% aperture shapes.
93resultGUI = matRad_siochiLeafSequencing(resultGUI,stf,dij,5);
94
95%% DAO - Direct Aperture Optimization
96% The Direct Aperture Optimization is an optimization approach where we 
97% directly optimize aperture shapes and weights.
98resultGUI = matRad_directApertureOptimization(dij,cst,resultGUI.apertureInfo,resultGUI,pln);
99
100%% Aperture visualization
101% Use a matrad function to visualize the resulting aperture shapes
102matRad_visApertureInfo(resultGUI.apertureInfo);
103
104%% Indicator Calculation and display of DVH and QI
105[dvh,qi] = matRad_indicatorWrapper(cst,pln,resultGUI);
