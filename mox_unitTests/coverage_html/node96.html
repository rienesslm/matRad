177function [tops, bases] = matRad_siochiDecomposePort(map,dimZ,dimX,minZ,maxZ,minX,maxX)
178%Returns tops and bases of a fluence matrix "map" for Siochi leaf
179%sequencing algorithm (rod pushing part).  Accounts for collisions and
180%tongue and groove (Tng) effects.
181
182tops = zeros(dimZ, dimX);
183bases = zeros(dimZ, dimX);
184
185for i = minX:maxX
186    maxTop = -1;
187    TnG = 1;
188    for j = minZ:maxZ
189        if i == minX
190            bases(j,i) = 1;
191            tops(j,i) = bases(j,i)+map(j,i)-1;
192        else %assign trial base positions
193            if map(j,i) >= map(j,i-1) %current rod >= previous, match the bases
194                bases(j,i) = bases(j,i-1);
195                tops(j,i) = bases(j,i)+map(j,i)-1;
196            else %current rod <previous
197                if map(j,i) == 0 %rod length=0, put in in next slab after top of previous
198                    bases(j,i) = tops(j,i-1)+1;
199                    tops(j,i) = bases(j,i)-1;
200                else %rod length~=0, match tops
201                    tops(j,i) = tops(j,i-1);
202                    bases(j,i) = tops(j,i)-map(j,i)+1;
203                end
204            end
205        end
206        %determine which rod has the highest top in column
207        if tops(j,i) > maxTop
208            maxTop = tops(j,i);
209            maxRow = j;
210        end
211    end
212    
213    %Correct for collision and tongue and groove error
214    while(TnG)
215        %go from maxRow down checking for TnG.  This occurs when a shorter
216        %rod is "peeking over" a longer one in the direction transverse to
217        %the leaf motion.  To fix this, match either the tops or bases of
218        %the rods.
219        for j = (maxRow-1):-1:minZ
220            if map(j,i) < map(j+1,i)
221                if tops(j,i) > tops(j+1,i)
222                    tops(j+1,i) = tops(j,i);
223                    bases(j+1,i) = tops(j+1,i)-map(j+1,i)+1;
224                elseif bases(j,i) < bases(j+1,i)
225                    bases(j,i) = bases(j+1,i);
226                    tops(j,i) = bases(j,i)+map(j,i)-1;
227                end
228            else
229                if tops(j,i) < tops(j+1,i)
230                    tops(j,i) = tops(j+1,i);
231                    bases(j,i) = tops(j,i)-map(j,i)+1;
232                elseif bases(j,i) > bases(j+1,i)
233                    bases(j+1,i) = bases(j,i);
234                    tops(j+1,i) = bases(j+1,i)+map(j+1,i)-1;
235                end
236            end
237        end
238        %go from maxRow up checking for TnG
239        for j = (maxRow+1):maxZ
240            if map(j,i) < map(j-1,i)
241                if tops(j,i) > tops(j-1,i)
242                    tops(j-1,i) = tops(j,i);
243                    bases(j-1,i) = tops(j-1,i)-map(j-1,i)+1;
244                elseif bases(j,i) < bases(j-1,i)
245                    bases(j,i) = bases(j-1,i);
246                    tops(j,i) = bases(j,i)+map(j,i)-1;
247                end
248            else
249                if tops(j,i) < tops(j-1,i)
250                    tops(j,i) = tops(j-1,i);
251                    bases(j,i) = tops(j,i)-map(j,i)+1;
252                elseif bases(j,i) > bases(j-1,i)
253                    bases(j-1,i) = bases(j,i);
254                    tops(j-1,i) = bases(j-1,i)+map(j-1,i)-1;
255                end
256            end
257        end
258        %now check if all TnG conditions have been removed
259        TnG = 0;
260        for j = (minZ+1):maxZ
261            if map(j,i) < map(j-1,i);
262                if tops(j,i) > tops(j-1,i)
263                    TnG = 1;
264                elseif bases(j,i) < bases(j-1,i)
265                    TnG = 1;
266                end
267            else
268                if tops(j,i) < tops(j-1,i)
269                    TnG = 1;
270                elseif bases(j,i) > bases(j-1,i)
271                    TnG = 1;
272                end
273            end
274        end
275    end
276end

346function diffSlab = matRad_siochiDifferentSlab(tops,bases,level)
347%Returns 1 if slab level is different than slab level-1 0 otherwise
348
349if level == 1 %first slab is automatically different
350    diffSlab = 1;
351else
352    shapeLevel = (bases <= level).*(level <= tops); %shape of slab with current level
353    shapeLevel_1 = (bases <= level-1).*(level-1 <= tops); %shape of slab with previous level
354    diffSlab = ~isequal(shapeLevel,shapeLevel_1); %tests if slabs are equal; isequaln was not giving correct results
355end
